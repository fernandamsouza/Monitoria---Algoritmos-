#include <stdlib.h>
#include<stdio.h>
#define TAM 20

int main(){
    //Quantidade de números
    //Vetor com os números
    int vet[TAM];
    //Obtém a lista
    int i;
    for(i = 0;i<TAM;i++){
        printf("\n\nDigite o numero para a posicao %d: ",i);
        scanf("%d",&vet[i]);
    }
    //Percorre os números e armazena os repetidos
    printf("\n");
    int j, cont;
    for(i = 0; i < TAM; i++){
        cont=0;
        for(j = 0; j < TAM; j++){
            if(vet[j]==vet[i]){
                if(j<i){
                    //essa verificação já foi feita antes, portanto o valor já foi escrito na tela
                    break;
                }
                cont++;
            }
        }
        if(cont>1){
            printf("O numero %d se repete %d vezes\n",vet[i],cont);
        }
}
    system("pause");
    return 0;
}
